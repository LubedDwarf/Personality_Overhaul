# 5001: RECIPROCAL OBLIGATION
pop_dynamic_events.5001 = {
    type = character_event
    title = pop_dynamic_events.5001.t
    desc = pop_dynamic_events.5001.d
    theme = diplomacy
    left_portrait = root
    right_portrait = scope:obligation_proposer

    option = {
        name = pop_dynamic_events.5001.a
        trigger = {
            custom_description = {
                text = "pop_willpower_gated_by_erratic_nature_tooltip"
                not = {
                    root = { pop_erratic_score > 40 }
                    not = { root = { has_variable = pop_willpower_points check_variable = { name = pop_willpower_points value >= 3 } } }
                }
            }
        }
        ai_chance = {
            base = 0
            modifier = { add = root.pop_dutiful_score }
            modifier = { add = root.pop_devoted_score }
        }
        if = {
            limit = { root.pop_erratic_score > 40 }
            change_variable = { name = pop_willpower_points value = -2 }
            add_stress = 60
        }
        remove_stress = 15
        scope:obligation_proposer = { add_prestige = 50 }
    }
    option = {
        name = pop_dynamic_events.5001.b
        ai_chance = {
            base = 10
            modifier = { add = root.pop_cunning_score }
            modifier = { add = root.pop_resolute_score }
        }
        scope:obligation_proposer = { pop_modify_leverage = { TARGET = root CHANGE = 30 } }
        add_stress = 25
    }
    option = {
        name = pop_dynamic_events.5001.c
        ai_chance = {
            base = 20
            modifier = { add = root.pop_compassionate_score }
            modifier = { add = root.pop_sociable_score }
        }
        remove_stress = 35
        scope:obligation_proposer = { remove_stress = 35 add_opinion = { target = root value = 30 } }
        add_opinion = { target = scope:obligation_proposer value = 30 }
    }
    option = {
        name = pop_dynamic_events.5001.d
        trigger = { root.pop_timid_score > 25 }
        ai_chance = { base = 100 }
        change_variable = { name = pop_willpower_points value = -1 }
        scope:obligation_proposer = { pop_modify_leverage = { TARGET = root CHANGE = 40 } }
        add_stress = 10
    }
    option = {
        name = pop_dynamic_events.5001.e
        trigger = { or = { root.pop_erratic_score > 25 scope:obligation_proposer.pop_erratic_score > 25 } }
        ai_chance = { base = 150 }
        add_stress = 40
    }
}
