namespace = pop_evolution_events

# Core evolution event framework - triggers when personality shifts occur
pop_evolution_events.0001 = {
    type = character_event
    title = pop_evolution_events.0001.t
    desc = pop_evolution_events.0001.d
    theme = "family"

    left_portrait = root

    trigger = {
        has_character_flag = pop_trait_evolution_confirmed
    }

    immediate = {
        # Mark evolution as recorded for UI tooltips
        set_character_flag = pop_evolution_confirmed
    }

    option = {
        name = pop_evolution_events.0001.a
        add_character_flag = {
            flag = pop_trait_evolution_confirmed
            years = 1
        }
        add_stress = 10
        remove_trait = callous
        add_trait = legendary_strategy
    }
}
